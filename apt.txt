# Packages here supplement what the pinned base-python-image (the FROM base)
# already provides. Anything already in base -- including via Ubuntu 24.04 "t64"
# renamed packages (e.g. libcups2t64, libatspi2.0-0t64) -- is intentionally
# omitted to keep this list minimal. Verified against base aa924984d219.

# libzmq.so.5 for pbdZMQ -> IRkernel::installspec (base has no system libzmq).
libzmq5

# System libraries for the geospatial R packages (sf, terra, stars, gdalcubes,
# mapgl, rstac). Base provides none of libgdal/libgeos/libproj/libudunits2.
libgdal-dev
libgeos-dev
libproj-dev
libudunits2-dev

# libMagick++ for the R 'magick' package. Base ships MagickWand/Core but not
# the Magick++ C++ API that magick links against.
libmagick++-dev

# Real ALSA library for headless chromium (webpdf "Save and Export Notebook As
# -> PDF"). Base only ships the incomplete liboss4-salsa-asound2 shim, which
# lacks snd_device_name_get_hint@ALSA_0.9 and crashes chromium.
libasound2t64
